import { Tabs, TabItem } from "@astrojs/starlight/components";

## Creating a new PromethiumJS application

To create a new PromethiumJS application, run one of the following commands based on your preferred package manager.

<Tabs>
  <TabItem label="npm">
  
  ```sh
  npm create promethium@latest 
  ```
  
  </TabItem>
  <TabItem label="yarn">

```sh
yarn create promethium
```

  </TabItem>
  <TabItem label="pnpm">

```sh
pnpm create promethium
```

  </TabItem>
  <TabItem label="bun">

```sh
bunx create-promethium
```

  </TabItem>
</Tabs>

## Adding PromethiumJS to an existing Lit application

To add PromethiumJS to an existing Lit application, run one of the following commands based on your preferred package manager.

<Tabs>
  <TabItem label="npm">
  
  ```sh
  npm install promethium-js 
  ```
  
  </TabItem>
  <TabItem label="yarn">

```sh
yarn add promethium-js
```

  </TabItem>
  <TabItem label="pnpm">

```sh
pnpm install promethium-js
```

  </TabItem>
  <TabItem label="bun">

```sh
bunx install promethium-js
```

  </TabItem>
</Tabs>

:::note
You can add PromethiumJS to a lit-html application this way as well.
:::
